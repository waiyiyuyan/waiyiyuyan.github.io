<script>
    // 增加一个延时处理，确保DOM加载完成
    window.onload = function() {
        fetch('https://api.ip.sb/geoip')
            .then(response => response.json())
            .then(data => {
                document.getElementById('ip').innerText = data.ip;
                document.getElementById('loc').innerText = data.country + ' - ' + (data.region || '');
                document.getElementById('isp').innerText = data.isp;
            })
            .catch(err => {
                console.error(err);
                document.getElementById('ip').innerText = "获取失败，请刷新页面";
            });
    };
</script>
