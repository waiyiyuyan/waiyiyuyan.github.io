简介：全程基于 Cloudflare 免费资源搭建，无需服务器、无需备案、纯免费部署。集成 D1 数据库、WebSocket 实时通讯、DurableObject 全局会话、Imgur 全球图床，完美解决国内访问问题，支持文字、图片、视频留言、楼中楼回复、用户通知、分页防卡顿，是一套完整可商用的轻量化留言系统。

核心优势：

  • ✅ 全自动实时更新，无需刷新页面
  • ✅ 游标分页加载，万条数据不卡顿，彻底解决全量查询压力
  • ✅ 支持图片、MP4/WEBM 视频留言，自动修复视频强制下载问题
  • ✅ 完整@用户通知、上线自动补发离线未读消息
  • ✅ 双 Worker 架构，分离上传业务与资源代理，稳定不冲突
  • ✅ 全部跨域配置内置，无需手动处理跨域报错
  • ✅ 国内可正常访问 Imgur 图片视频，解决境外资源拦截问题

一、整体架构认知（必看，杜绝配置混乱）

整套系统由 3 个独立服务 组成，各司其职、互不干扰，所有配置、域名、修改位置完全分开：

1. 前端服务（Cloudflare Pages）

托管于 GitHub + Cloudflare Pages，纯静态页面，无任何后端逻辑。仅负责页面展示、用户操作、调用后端接口。

✅ 需配置：2 个后端服务域名
✅ 绑定：前端展示域名（你的留言板访问网址）

2. 主业务后端（Cloudflare Worker）

系统核心服务，绑定 D1 数据库 + DurableObject。承担数据库读写、WebSocket 实时推送、发帖/回复/删除/通知、Imgur 文件上传全部核心业务。

✅ 需配置：D1 数据库信息、跨域白名单、Imgur 密钥
✅ 绑定：后端 API 域名

3. Imgur 资源代理后端（Cloudflare Worker）

独立中转服务，专门解决国内无法访问 Imgur 原图、视频强制下载、图片跨域报错问题。24小时缓存资源，加速访问。

✅ 零代码配置，粘贴即用
✅ 绑定：图片视频代理域名

function add8h(utcStr) {
  if (!utcStr) return "";
  const date = new Date(utcStr);
  const bjTime = new Date(date.getTime() + 8 * 60 * 60 * 1000);
  const year = bjTime.getFullYear();
  const month = String(bjTime.getMonth() + 1).padStart(2, "0");
  const day = String(bjTime.getDate()).padStart(2, "0");
  const h = String(bjTime.getHours()).padStart(2, "0");
  const m = String(bjTime.getMinutes()).padStart(2, "0");
  const s = String(bjTime.getSeconds()).padStart(2, "0");
  return `${year}-${month}-${day} ${h}:${m}:${s}`;
}

async function queryFullPosts(env) {
  try {
    const { results: posts } = await env.MSG_DB.prepare(
      "SELECT * FROM message ORDER BY create_time DESC"
    ).all();
    const { results: allReply } = await env.MSG_DB.prepare(
      "SELECT * FROM reply ORDER BY create_time ASC"
    ).all();
    const out = [];
    for (let p of posts) {
      let replys = allReply.filter(item => item.msg_id === p.id);
      p.create_time = add8h(p.create_time);
      replys.forEach(item => item.create_time = add8h(item.create_time));

  if (!uid) return [];
  const { results: notifyList } = await env.MSG_DB.prepare(

  return notifyList;
}

export class ChatRoom extends DurableObject {
  constructor(state, env) {
    super(state, env);
    this.state = state;
    this.env = env;
    this.sessions = new Set();
  }

  sendToOne(ws, data) {
    try {
      ws.send(JSON.stringify(data));
    } catch (err) {
      this.sessions.delete(ws);
    }
  }

  broadcast(data) {
    const msg = JSON.stringify(data);
    for (const conn of this.sessions) {
      try {
        conn.send(msg);
      } catch (err) {
        this.sessions.delete(conn);
      }
    }
  }

  async sendNotifyToUser(uid, notifyList) {
    for (const conn of this.sessions) {
      if (conn.userUid === uid) {
        try {
          await conn.send(JSON.stringify({
            type: "NOTIFY_DATA",
            notify: notifyList
          }));
          return;
        } catch (err) {
          this.sessions.delete(conn);
        }
      }
    }
  }

  broadcastLog(level, content) {
    try {
      const msg = JSON.stringify({ type: "SYS_LOG", level, content });
      for (const conn of this.sessions) conn.send(msg);
    } catch (err) {}
  }

  async fetch(request) {
    const [clientWs, serverWs] = new WebSocketPair();
    serverWs.accept();
    this.sessions.add(serverWs);

    serverWs.addEventListener("close", () => {
      this.sessions.delete(serverWs);
    });

    serverWs.addEventListener("open", async () => {
      try {
        const initPosts = await queryFullPosts(this.env);
        this.sendToOne(serverWs, { type: "INIT_DATA", posts: initPosts, notify: [] });
      } catch (e) {}
    });

    serverWs.addEventListener("message", async (event) => {
      try {
        const msg = JSON.parse(event.data);
        if (msg.type === "HEARTBEAT") return;
        if (msg.type === "USER_UID" && msg.uid) {
          const uid = msg.uid;
          serverWs.userUid = uid;
          const userNotify = await queryUserNotify(this.env, uid);
          await this.sendToOne(serverWs, { type: "NOTIFY_DATA", notify: userNotify });
        }
      } catch (e) {}
    });

    return new Response(null, { status: 101, webSocket: clientWs });
  }
}

async function queryPostsByCursor(env, cursor = "") {
  let posts = [];
  if (!cursor) {
    const { results } = await env.MSG_DB.prepare(
      "SELECT * FROM message ORDER BY id DESC LIMIT 10"
    ).all();
    posts = results;
  } else {
    const { results } = await env.MSG_DB.prepare(
      "SELECT * FROM message WHERE id < ? ORDER BY id DESC LIMIT 10"
    ).bind(cursor).all();
    posts = results;
  }
  if (posts.length === 0) return { list: [], lastCursor: "", hasMore: false };
  const postIds = posts.map(p => p.id);
  const placeholders = postIds.map(() => "?").join(",");
  const { results: allReply } = await env.MSG_DB.prepare(
    `SELECT * FROM reply WHERE msg_id IN (${placeholders}) ORDER BY create_time ASC`
  ).bind(...postIds).all();
  const out = [];
  for (let p of posts) {
    let replys = allReply.filter(item => item.msg_id === p.id);
    p.create_time = add8h(p.create_time);
    replys.forEach(item => item.create_time = add8h(item.create_time));
    out.push({ ...p, replys });
  }
  const lastCursor = out[out.length - 1].id;
  const hasMore = out.length === 10;
  return { list: out, lastCursor, hasMore };
}

async function getPostDetailById(env, postId) {
  if (!postId) return null;
  const { results: postList } = await env.MSG_DB.prepare(
    "SELECT * FROM message WHERE id = ? LIMIT 1"
  ).bind(postId).all();
  if (postList.length === 0) return null;
  const post = postList[0];
  const { results: replyList } = await env.MSG_DB.prepare(
    "SELECT * FROM reply WHERE msg_id = ? ORDER BY create_time ASC"
  ).bind(postId).all();
  post.create_time = add8h(post.create_time);
  replyList.forEach(item => item.create_time = add8h(item.create_time));
  return { ...post, replys: replyList };
}

async function getLatestSinglePost(env) {

  if (results.length === 0) return null;
  const post = results[0];
  post.create_time = add8h(post.create_time);
  post.replys = [];
  return post;
}

    }

    // 初始化分页数据
    if (url.pathname === "/initPosts" && req.method === "GET") {
      try {
        const result = await queryPostsByCursor(env);
        return Response.json({ code: 0, list: result.list, lastCursor: result.lastCursor, hasMore: result.hasMore }, { headers: corsHeaders });
      } catch (e) {
        return Response.json({ code: -1, msg: "查询失败", list: [], lastCursor: "", hasMore: false }, { status: 500, headers: corsHeaders });
      }
    }

    // 加载更多
    if (url.pathname === "/loadMorePosts" && req.method === "GET") {
      try {
        const cursor = url.searchParams.get("cursor") || "";
        if (!cursor) return Response.json({ code: -1, msg: "缺少游标参数", list: [], lastCursor: "", hasMore: false }, { headers: corsHeaders });
        const result = await queryPostsByCursor(env, cursor);
        return Response.json({ code: 0, list: result.list, lastCursor: result.lastCursor, hasMore: result.hasMore }, { headers: corsHeaders });
      } catch (e) {
        return Response.json({ code: -1, msg: "查询失败", list: [], lastCursor: "", hasMore: false }, { status: 500, headers: corsHeaders });
      }
    }

    // 帖子详情
    if (url.pathname === "/getPostDetail" && req.method === "GET") {
      try {
        const postId = url.searchParams.get("id") || "";
        if (!postId) return Response.json({ code: -1, msg: "缺少帖子ID", data: null }, { headers: corsHeaders });
        const postDetail = await getPostDetailById(env, postId);
        return Response.json({ code: 0, msg: "ok", data: postDetail || null }, { headers: corsHeaders });
      } catch (e) {
        return Response.json({ code: -1, msg: "查询失败", data: null }, { status: 500, headers: corsHeaders });
      }
    }

    // 兼容旧接口
    if (url.pathname === "/listAll") return Response.json(await queryFullPosts(env), { headers: corsHeaders });
    if (url.pathname === "/getNotify") {
      const uid = url.searchParams.get("uid") || "";
      return Response.json(await queryUserNotify(env, uid), { headers: corsHeaders });
    }

    if (url.pathname === "/") return new Response("API Service Running", { headers: { ...corsHeaders, "Content-Type": "text/plain;charset=UTF-8" } });
    return new Response("404 Not Found", { status: 404, headers: { ...corsHeaders, "Content-Type": "text/plain;charset=UTF-8" } });
  }
};

五、第三步：部署 Imgur 资源代理 Worker（独立服务）

新建空白 Worker，全量复制以下代码，无需任何修改，直接部署，部署后绑定独立代理域名。

export default {
  async fetch(request) {
    const corsHeaders = {
      "Access-Control-Allow-Origin": "*",
      "Cache-Control": "public, max-age=86400"
    };
    if (request.method === "OPTIONS") return new Response(null, { status: 204, headers: corsHeaders });
    if (request.method !== "GET") return new Response("Method Not Allowed", { status: 405, headers: corsHeaders });
    const url = new URL(request.url);
    const mediaUrl = url.searchParams.get("url");
    if (!mediaUrl || !mediaUrl.startsWith("https://i.imgur.com/")) return new Response("Invalid URL", { status: 400, headers: corsHeaders });
    try {
      const res = await fetch(mediaUrl, { headers: { "User-Agent": request.headers.get("User-Agent") || "" } });
      const newHeaders = new Headers(res.headers);
      const lowerUrl = mediaUrl.toLowerCase();
      if (lowerUrl.endsWith(".mp4") || lowerUrl.endsWith(".webm")) {
        newHeaders.delete("Content-Disposition");
        newHeaders.delete("content-disposition");
        newHeaders.set("Content-Type", lowerUrl.endsWith(".mp4") ? "video/mp4" : "video/webm");
        newHeaders.set("Accept-Ranges", "bytes");
      }
      Object.entries(corsHeaders).forEach(([k, v]) => newHeaders.set(k, v));
      return new Response(res.body, { status: res.status, headers: newHeaders });
    } catch (err) {
      return new Response("Media load failed", { status: 502, headers: corsHeaders });
    }
  }
};

八、核心功能原理说明

1. 分页防卡顿原理

摒弃一次性全量查询数据库，采用游标分页，每次仅加载10条最新留言。无论数据库存储多少数据，单用户单次请求仅查询10条，彻底解决高并发、大数据量页面卡死问题。

2. 实时通讯原理

基于 Cloudflare DurableObject 维持全局唯一会话，所有在线用户共享 WebSocket 连接，发帖、回复、删除操作实时广播，无需刷新页面。用户离线后再次上线，自动补发未读@通知。

3. Imgur 兼容方案

上传逻辑集成主后端，通过 Worker 代理请求 Imgur 官方接口，规避本地网络拦截；图片视频展示通过独立代理 Worker 中转，修复视频强制下载问题、解决国内境外资源无法访问问题，24小时缓存加速访问。

九、常见问题排错手册

  • 图片空白加载失败：检查前端 PROXY_PREFIX 域名及后缀格式是否正确，必须带 /?url=
  • 视频点击直接下载：必须使用代理 Worker 链接，禁止直接使用 Imgur 原生链接
  • 上传图片失败：检查主 Worker 内 Imgur ClientID 是否正确、有效
  • 跨域报错：检查后端跨域白名单是否填写完整前端+后端域名
  • 无实时更新：检查主 Worker 是否正常部署、WebSocket 路由正常、DO 绑定生效
  • 数据库读写失败：检查 wrangler.jsonc 内 D1 数据库ID、名称是否和云端一致
